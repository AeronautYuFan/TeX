    $$f(x) = \begin{cases}
    \frac{1}{x} & x \neq 0\\
    0 & x = 0
    \end{cases}.$$
    \end{enumerate}
    Does $f$ satisfy $p$ or $\neg p$? Prove your answer.
    

\end{problem}
